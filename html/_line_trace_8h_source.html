226  double mScale = 1.0f; // When different from 1.0, LSD will scale the input image
227  //by 'scale' factor by Gaussian filtering, before detecting line segments.
228  //Example: if scale = 0.8, the input image will be subsampled
229  //to 80 % of its size, before the line segment detector
230  //is applied. Suggested value : 0.8
231  double mSigmaScale = 0.6f; //When scale != 1.0, the sigma of the Gaussian filter is :
232  //sigma = sigma_scale / scale, if scale < 1.0
233  //sigma = sigma_scale, if scale >= 1.0
234  //Suggested value : 0.6
235  double mAngleThr = 22.5f; // Gradient angle tolerance in the region growing algorithm, in degrees.
236  //Suggested value : 22.5
237  double mLogEps = 0.0f; //Detection threshold, accept if - log10(NFA) > log_eps.
238  // The larger the value, the more strict the detector is,
239  // and will result in less detections.
240  // (Note that the 'minus sign' makes that this
241  // behavior is opposite to the one of NFA.)
242  // The value - log10(NFA) is equivalent but more
243  // intuitive than NFA :
244  //--1.0 gives an average of 10 false detections on noise
245  // - 0.0 gives an average of 1 false detections on noise
246  // - 1.0 gives an average of 0.1 false detections on nose
247  // - 2.0 gives an average of 0.01 false detections on noise
248  // .
249  // Suggested value : 0.0
250  double mDensityThr = 0.7f; //Minimal proportion of 'supporting' points in a rectangle. Suggested value : 0.7
251  int mNBins = 1024; //Number of bins used in the pseudo-ordering of gradient modulus. Suggested value : 1024
