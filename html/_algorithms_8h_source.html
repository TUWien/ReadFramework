101  template <typename numFmt>
102  static double statMoment(const QList<numFmt>& valuesIn, double momentValue, int interpolated = 1) {
103 
104  QList<numFmt> values = valuesIn;
105  qSort(values);
106 
107  size_t lSize = values.size();
108  double moment = -1;
109  unsigned int momIdx = cvCeil(lSize*momentValue);
110  unsigned int idx = 1;
111 
112  // find the statistical moment
113  for (auto val : values) {
114 
115  // skip
116  if (idx < momIdx) {
117  idx++;
118  continue;
119  }
120  if (lSize % 2 == 0 && momIdx < lSize && interpolated == 1)
121  // compute mean between this and the next element
122  moment = ((double)val + values[idx+1])*0.5;
123  else
124  moment = (double)val;
125  break;
126  }
127 
128  return moment;
129  }
